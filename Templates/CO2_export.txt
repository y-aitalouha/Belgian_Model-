#TIMEHORIZON
T = 8760;

#NODE CO2_EXPORT
// data from Berger et al. 2019

#PARAMETERS
energy_carbon_dioxide = 0;
export_cost = 2 * 1/(10**3);                    // M€/kt
export_capacity = 3.5;                          // kt/h

#VARIABLES
external: co2_exported[T];                          // kt 

#CONSTRAINTS
co2_exported[t] >= 0;
co2_exported[t] <= export_capacity;

#OBJECTIVES
min export_cost: export_cost * co2_exported[t];     // M€
